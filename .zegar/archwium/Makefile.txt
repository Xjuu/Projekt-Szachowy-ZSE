CXX := g++
CXXFLAGS := -Wall -Wextra -O2 -std=c++17 $(shell sdl2-config --cflags)
LDFLAGS := $(shell sdl2-config --libs) -lSDL2_ttf

SRC := src/main.cpp src/clock.cpp src/ui.cpp
OUT := build/chess_clock

.PHONY: all clean

all: $(OUT)

$(OUT): $(SRC)
	@mkdir -p build
	$(CXX) $(CXXFLAGS) $(SRC) -o $(OUT) $(LDFLAGS)

clean:
	rm -rf build
